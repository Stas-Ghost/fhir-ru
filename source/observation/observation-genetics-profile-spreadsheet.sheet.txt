=== Sheet ObservationForGenetics =====================================

 -- Row 1 -----------------------------------
  Element = Observation
  Profile Name = Genetics

 -- Row 2 -----------------------------------
  Element = Observation.extension
  Profile Name = TranscriptReferenceSequenceId
  Card. = 0..1
  Type = Extension{#TranscriptReferenceSequenceId}
  Short Label = Transcript Reference Sequence Identifier
  Definition = This field carries the ID for the transcribed reference sequence that part of the genetic reference sequence that is converted to Messenger RNA.

 -- Row 3 -----------------------------------
  Element = Observation.extension
  Profile Name = ProteinReferenceSequenceId
  Card. = 0..1
  Type = Extension{#ProteinReferenceSequenceId}
  Short Label = Protein Reference Sequence
  Definition = Reference identifier for protein transcript, with version, from NCBI\'s RefSeq or ENSEMBL.

 -- Row 4 -----------------------------------
  Element = Observation.extension
  Profile Name = DNASequenceVariation
  Card. = 0..1
  Type = Extension{#DNASequenceVariation}
  Short Label = DNA Sequence Variation
  Definition = Human Genome Variation Society (HGVS) nomenclature for a single or set of DNA Sequence Variation(s) identified in testing. Reference Sequence id used for HGVS naming must be anotated.

 -- Row 5 -----------------------------------
  Element = Observation.extension
  Profile Name = DNAVariationId
  Card. = 0..1
  Type = Extension{#DNAVariationId}
  Short Label = Variant Id
  Definition = Identifier for variant.  If a germline variant, ClinVar or dbSNP identifier should be used.  If a somatic variant, COSMIC identifier should be used, unless in ClinVar then either maybe used. Need to provide the code system used (ClinVar, dbSNP, COSMIC)

 -- Row 6 -----------------------------------
  Element = Observation.extension
  Profile Name = DNASequenceVariationType
  Card. = 0..1
  Type = Extension{#DNASequenceVariationType}
  Short Label = DNA Variant Type
  Definition = Codified type for associated DNA Sequence Variation. DNA Sequence Variations use the HGVS notation which implies the DNA Sequence Variation Type

 -- Row 7 -----------------------------------
  Element = Observation.extension
  Profile Name = AminoAcidChange
  Card. = 0..1
  Type = Extension{#AminoAcidChange}
  Short Label = Amino Acid Change
  Definition = Human Genome Variation Society (HGVS) nomenclature for an amino acid change. Reference Sequence id used for HGVS naming must be anotated.
  To Do = Need OID for SO

 -- Row 8 -----------------------------------
  Element = Observation.extension
  Profile Name = AminoAcidChangeType
  Card. = 0..1
  Type = Extension{#AminoAcidChangeType}
  Short Label = Amino Acid Change Type
  Definition = Codified type for associated Amino Acid Change.

 -- Row 9 -----------------------------------
  Element = Observation.extension
  Profile Name = Gene
  Card. = 0..1
  Type = Extension{#Gene}
  Short Label = Gene Identifier
  Definition = Gene in which the sequence is found. Currently values from HGNC are supported. Other systems or genes not defined in HGNC (e.g. BCR-ABL fusion gene) can be added by using local extension.

 -- Row 10 -----------------------------------
  Element = Observation.extension
  Profile Name = DNARegionName
  Card. = 0..1
  Type = Extension{#DNARegionName}
  Short Label = DNA Region Name
  Definition = A human readable name for the region of interest. Typically Exon #, Intron # or other. NOTE: This is not standardized and is mainly for convenience and display purposes.
  To Do = Need OID for SO

 -- Row 11 -----------------------------------
  Element = Observation.extension
  Profile Name = AlleleName
  Card. = 0..1
  Type = Extension{#AlleleName}
  Short Label = AlleleName
  Definition = Common name for variant or gene allele

 -- Row 12 -----------------------------------
  Element = Observation.extension
  Profile Name = GenomicSourceClass
  Card. = 0..1
  Type = Extension{#GenomicSourceClass}
  Short Label = Genomic Source Class
  Definition = Source of sample used to determine the sequence in sequencing lab -- germline, somatic, prenatal.

 -- Row 13 -----------------------------------
